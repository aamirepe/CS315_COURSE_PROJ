# Simple SELECT Operations (No Joins)
students|SELECT * FROM students
grades|SELECT * FROM grades
enrollments|SELECT * FROM enrollments
TC1|SELECT * FROM students WHERE id=5
TC2|SELECT * FROM students WHERE age > 20
TC3|SELECT * FROM courses
TC4|SELECT * FROM courses WHERE id = 2

# Selection Push-Down (Single Join)
TC5|SELECT * FROM students JOIN grades ON students.id = grades.student_id WHERE students.age = 18

# Joins
TC6|SELECT * FROM students JOIN enrollments ON students.id = enrollments.student_id
TC7|SELECT * FROM students JOIN enrollments ON students.id = enrollments.student_id JOIN courses ON enrollments.course_id = courses.id
TC8|SELECT * FROM students JOIN grades ON students.id = grades.student_id JOIN enrollments ON students.id = enrollments.student_id JOIN courses ON enrollments.course_id = courses.id WHERE students.age > 18
